country_event = {#choice to either pay local landholders to back down from the rebellion or spawn rebels
	id = uis_ideagroups.4001
	title = uis_ideagroups.4001.t
	desc = uis_ideagroups.4001.d
	picture = SIEGE_eventPicture
	
	is_triggered_only = yes
	trigger = {	has_disaster = uis_militarism_junta	}
	
	option = {#Fight the traitors
		name = uis_ideagroups.4001.a
		add_prestige = 10
		add_yearly_manpower = -0.25
		add_stability = -1
		
		hidden_effect = {
			change_variable = {
				which = junta_progress
				value = 4
			}
		}
		#should spawn rebels on ONE fort province that is not the capital and immediately occupy it fallback to any random province.
		if = {
			limit = { 
				any_owned_province = {
					fort_level = 2
					is_capital = no
				}
			}
			random_owned_province = {
				limit = {
					fort_level = 2
					is_capital = no				
				}
				spawn_rebels = {
					size = 2
					type = junta_rebels
					win = yes
					should_take_capital = yes
				}
			}
			
		}
		else = {
			random_owned_province = {
				limit = {
					is_capital = no
				}
				spawn_rebels = {
					size = 2
					type = junta_rebels
					win = yes
					should_take_capital = yes
				}
			}
		}
	}
	option = {#pay off traitors with 1 year of income and some corruption gain.
		name = uis_ideagroups.4001.b
		add_corruption = 2
		add_years_of_income = -2
		add_country_modifier = {
			name = uis_junta_forts_paid
			duration = -1
		}
		hidden_effect = {
			change_variable = {
				which = junta_progress
				value = 9
			}
		}	
	}
	after = {
		if = {
			limit = {
				OR = {
					AND = {
						has_estate = estate_nobles
						NOT = {
							estate_loyalty = {
								estate = estate_nobles
								loyalty = 60
							}
						}
					}
					AND = {
						has_estate = estate_rajput
						NOT = {
							estate_loyalty = {
								estate = estate_rajput
								loyalty = 60
							}
						}
					}
					AND = {
						has_estate = estate_maratha
						NOT = {
							estate_loyalty = {
								estate = estate_maratha
								loyalty = 60
							}
						}
					}
				}
			}
			country_event = {
				id = uis_ideagroups.4003
				days = 120
				random = 30
			}
		}
		else = {
			country_event = {
				id = uis_ideagroups.4004
				days = 120
				random = 30
			}
		}
	}
