country_decisions = {
	ai_hold_court = { # Court
		potential = {
			ai = yes
		}
		allow = {
			ai = yes
			full_idea_group = court_ideas
			NOT = {
				AND = {
					has_ruler_flag = court_realm_decree_taken
					has_ruler_flag = court_economy_decree_taken
					has_ruler_flag = court_diplomacy_decree_taken
					has_ruler_flag = court_military_decree_taken
				}
			}
		}
		effect = {
			country_event = {
				id = ai_only_events.0001
			}
		}
		ai_will_do = {
			factor = 100
		}
	}

	ai_take_assimilation_reform = { # Assimilation
		potential = {
			ai = yes
		}
		allow = {
			ai = yes
			full_idea_group = assimilation_ideas
			government_reform_progress = 50
			num_of_owned_provinces_with = {
				value = 15
				has_owner_accepted_culture = no
				NOT = { culture = root }
			}
		}
		effect = {
			add_government_reform = assimilation_reform
			change_government_reform_progress = -50
		}
		ai_will_do = {
			factor = 100
		}
	}

	ai_switch_tactics = { # Tactics
		potential = {
			ai = yes
		}
		allow = {
			ai = yes
			full_idea_group = tactic_ideas
			NOT = { has_country_modifier = tactics_cooldown }
		}
		effect = {
			country_event = {
				id = ai_only_events.0002
			}
		}
		ai_will_do = {
			factor = 100
		}
	}

	ai_draft_pop = { # Quantity
		potential = {
			ai = yes
		}
		allow = {
			ai = yes
			full_idea_group = quantity_ideas
		}
		effect = {
			random_owned_province = {
				limit = {
					controlled_by = owner
					NOT = { has_province_modifier = pops_drafted }
				}
				set_province_flag = pops_drafted_flag
				owner = {
					add_years_of_owned_provinces_manpower = {
						years = 2
						trigger = {
							has_province_flag = pops_drafted_flag
						}
					}
				}
				clr_province_flag = pops_drafted_flag
				add_permanent_province_modifier = {
					name = pops_drafted
					duration = 1826
				}
			}
		}
		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				is_at_war = no
			}
			modifier = {
				factor = 0
				army_professionalism = 0.25
			}
			modifier = {
				factor = 0
				manpower_percentage = 0.25
			}
			modifier = {
				factor = 0
				NOT = { war_exhaustion = 5 }
			}
			modifier = {
				factor = 0
				war_score = -25
			}
		}
	}

	ai_training_interface = { # Training Mechanic
		potential = {
			ai = yes
		}
		allow = {
			ai = yes
			if = {
				limit = {
					check_variable = {
						which = trainingExperience
						value = 75
					}
					NOT = {
						OR = {
							# Inf

							# Cav

							# Art
						}
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {
						which = trainingExperience
						value = 50
					}
					NOT = {
						OR = {
							# Inf

							# Cav

							# Art
						}
					}
				}
			}
			else_if = {
				limit = {
					check_variable = {
						which = trainingExperience
						value = 25
					}
					NOT = {
						OR = {
							# Inf
							has_country_flag = inf_tech1_tier1
							has_country_flag = inf_tech2_tier1
							has_country_flag = inf_tech3_tier1
							has_country_flag = inf_tech4_tier1
							# Cav
							has_country_flag = cav_tech1_tier1
							has_country_flag = cav_tech2_tier1
							has_country_flag = cav_tech3_tier1
							# Art
							has_country_flag = art_tech1_tier1
							has_country_flag = art_tech2_tier1
							has_country_flag = art_tech3_tier1
						}
					}
				}
			}
			check_variable = {
				which = trainingExperience
				value = 25
			}
			NOT = { has_country_flag = ai_training_interface_open }
		}
		effect = {
			country_event = {
				id = ai_only_events.0003
			}
		}
		ai_will_do = {
			factor = 100
		}
	}
}
