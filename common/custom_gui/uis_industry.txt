# Prospecting button
custom_button = {
	name = prospect_for_resources
	tooltip = prospect_for_resources_TT
	potential = {
		owned_by = FROM
		controlled_by = FROM
	}
	trigger = {
		hidden_trigger = {
			owner = {
				full_idea_group = infrastructure_ideas
			}
		}
		OR = {
			has_terrain = hills
			has_terrain = highlands
			has_terrain = mountain
		}
	}
	effect = {
		industry_cost = yes
		add_province_modifier = {
			name = prospecting_for_resources
			duration = -1
		}
		hidden_effect = {
			province_event = {
				id = uis_ideagroups.3000
				days = 1825
				random = 182
			}
		}
	}
}

	name = industry_change_trade_good_button

	trigger = {
		always = yes
		# Disabled during development, enable before deploy
		#full_idea_group = infrastructure_ideas

	effect = {
		hidden_effect = {
			if = {
				limit = { has_country_flag = industry_change_trade_good_window_open }

			}
			else = {
				set_country_flag = industry_change_trade_good_window_open
			}
		}
	}
}
