# Helper effects
tech_tree_window_reset = {
	clr_country_flag = infantry_tree_window_shown
	clr_country_flag = cavalry_tree_window_shown
	clr_country_flag = artillery_tree_window_shown
}

# Initialize trainingExperience variable
initialize_training_variable = {
	set_variable = {
		which = trainingExperience
		value = 0
	}
}

# Add Experience
add_exp = {
	custom_tooltip = EXP_GAIN
	hidden_effect = {
		change_variable = {
			which = trainingExperience
			value = 1
		}
	}
}

# Subtract Experience
subtract_exp = {
	custom_tooltip = EXP_LOSS
	hidden_effect = {
		subtract_variable = {
			which = trainingExperience
			value = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				which = trainingExperience
				value = 0
			}
		}
		set_variable = {
			which = trainingExperience
			value = 0
		}
	}
}
